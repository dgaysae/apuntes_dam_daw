```java
import javax.swing.*;
import java.awt.*;

/**
 * Esta clase extiende de JButton para crear botones personalizados
 * 
 * @author Rubén J.
 */
public class BotonBonito extends JButton {

    /*
    Aributos de la clase que vamos a usar para crear botones
    personalizados con degragadados
    */
    private Color color1;
    private Color color2;

    /*
    Constructor que recibe el texto del boton y los 2 colores
    para el degradado
    */
    public BotonBonito(String texto, Color c1, Color c2) {
        super(texto);
        this.color1 = c1;
        this.color2 = c2;

        setFocusPainted(false);
        setContentAreaFilled(false);
        setBorderPainted(true);
        setForeground(Color.WHITE);
        setFont(new Font("Segoe UI Symbol", Font.BOLD, 25));
    }
    /*
    Sobreescribe el metodo paintComponent para pintar el degradado
    primero pinta el fondo con degradado y luego llama al super
    para el texto
    */
    @Override
    protected void paintComponent(Graphics g) {
        Graphics2D g2 = (Graphics2D) g;

        GradientPaint degradado = new GradientPaint(0, 0, color1, 0, getHeight(), color2);

        g2.setPaint(degradado);
        g2.fillRoundRect(0, 0, getWidth(), getHeight(), 20, 20);

        super.paintComponent(g);
    }
}
```

```java
    public void guardarOperador(String op) {
        if (!operador.isEmpty() && !operacionRealizada) {
            // si ya hay un operador guardado hace el cálculo
            calcular();

        } else {
            // guarda el número en pantalla casteandolo a double

        operador = op;

        jtpantalla.setText("");

        // limpia la pantalla
        jtpantalla.setText("");

        numero1 = 0;

        deja el operador a null porque supongo q si le das a la C
        es para empezar de nuevo una cuenta
        */
        operador = "";
        operacionRealizada = false;
    }

    /**
     * Muestra el resultado por pantalla y cambia la variable de
     * operacion realizada a true
     * @param resultado
     */
    public void mostrarResultado(double resultado) {
        jtpantalla.setText(String.valueOf(resultado));
        operacionRealizada = true;
    }

    /**
     * calcula de raiz cuadrada
     */
    public void calcularRaiz() {
        double numero = Double.parseDouble(textoActual());
        mostrarResultado(Math.sqrt(numero));
    }

    /**
     * Este método comprueba si ya tiene una coma antes de ponerla
     * y si no la tiene la pone y lo concatena a lo q ya había
     */
    public void añadirComa() {
        if (!textoActual().contains(".")) {
            jtpantalla.setText(textoActual() + ".");
        }
    }

    /**
     * Este método se encarga de borrar el último número de la
     * pantalla, lo borra a través del .substrinng el cual con
     * los parámetros (0, textoActual().lenght-1)
     * indicas que quieres quedarte en este caso indica que quiero
     * quedarme desde el primero hasta el último -1 (es decir
     * penúltimo), también chequea si el texto actual está vacío
     * si lo está reinicia toda la operación y borra el contenido
     * de la variable
     */
    public void borrarUltimo() {
        if (!textoActual().isEmpty()) {
            jtpantalla.setText(textoActual().substring(0, textoActual().length() - 1));

            if (textoActual().isEmpty()) {
                limpiarPantalla();
            }
        }

     * El objeto e que estamos metiendo no es más que un mensajero

     * botón se ha pulsado? ¿Cuándo? y que texto tenia Java te lo pasa
     * automáticamente al método para que puedas usarlo.
     * Tú lo llamas e simplemente para poder referirte a él dentro del
     * método.
     */
    public void actionPerformed(ActionEvent e) {
        String comando = e.getActionCommand();

        switch (comando) {
            case "0":
                concatenarNumero("0");
                break;
            case "1":
                concatenarNumero("1");
                break;
            case "2":
                concatenarNumero("2");
                break;
            case "3":
                concatenarNumero("3");
                break;
            case "4":
                concatenarNumero("4");
                break;
            case "5":
                concatenarNumero("5");
                break;
            case "6":
                concatenarNumero("6");
                break;
            case "7":
                concatenarNumero("7");
                break;
            case "8":
                concatenarNumero("8");
                break;
            case "9":
                concatenarNumero("9");
                break;

            case "+":
                guardarOperador("+");
                break;
            case "-":
                guardarOperador("-");
                break;
            case "/":
                guardarOperador("/");
                break;
            case "*":
                guardarOperador("*");
                break;
            case "%":
                guardarOperador("%");
                break;

            // llamamos al método calcular el cual se encarga de hacer la cuenta
            // correspondiente al operador guardado anteriormente
            case "=":
                calcular();
                break;

            case "C":
                limpiarPantalla();
                break;

            case "⌫":
                borrarUltimo();
                break;

            case "√":
                calcularRaiz();
                break;

            case ",":
                añadirComa();
                break;

            default:
                break;
        }
    }

    /**
     * Este método lo que hace es un switch donde segun el operador que hemos
     * guardado con el método guardar operador
     * recuperamos ese valor de la variable auxiliar y lo usamos para decidir la
     * operación, además usamos un método que es
     * mostrarResultado el cual muestra en la pantalla el resultado
     */
    public void calcular() {
        double numero2 = Double.parseDouble(textoActual());
        double resultado;

        switch (operador) {
            case "+":
                resultado = numero1 + numero2;
                mostrarResultado(resultado);
                break;
            case "-":
                resultado = numero1 - numero2;
                mostrarResultado(resultado);
                break;
            case "*":
                resultado = numero1 * numero2;
                mostrarResultado(resultado);
                break;
            case "/":
                // con este if sabemos si el número 2 es 0 entonces directamente no hacemos la
                // división, si no es el caso hacemos la división como de costumbre
                if (numero2 == 0) {
                    jtpantalla.setText("Error: div/0");
                } else {
                    resultado = numero1 / numero2;
                    mostrarResultado(resultado);
                }

                break;
            // me calcula el resto de la división de los números
            case "%":
                resultado = numero1 % numero2;
                mostrarResultado(resultado);
                break;

            default:
                break;
        }
    }

    /**
     * Ejecutar la calculadora
     * @param args
     */
    public static void main(String[] args) {
        interfaz_calculadora ic = new interfaz_calculadora();
    }
```

### 8.1.1. Botones

Para empezar, podemos simplificar la creación de botones. Hasta ahora vemos que para cada botón hay que introducir muchos parámetros y esto puede llevar a confusión y posibles errores.

Pero si creamos una clase botón para los números y otra para las operaciones, podríamos tener lo siguiente:

`BotonBonito.java`
```java
package view;

import javax.swing.*;
import java.awt.*;

/**
 * Establece las características básicas que deben tener todos los botones
 * de la aplicación.
 * No se puede instanciar directamente, ya que es abstracta.
 */
public abstract class BotonBonito extends JButton {
    public static final Color COLOR_OPER1 = new Color(80, 80, 80);
    public static final Color COLOR_OPER2 = new Color(30, 30, 30);
    public static final Color COLOR_NUMERO1 = new Color(30, 180, 150);
    public static final Color COLOR_NUMERO2 = new Color(10, 120, 100);

    private final Color color1;
    private final Color color2;

    public BotonBonito(String texto, Color c1, Color c2) {
        super(texto);
        this.color1 = c1;
        this.color2 = c2;

        setFocusPainted(false);
        setContentAreaFilled(false);
        setBorderPainted(true);
        setForeground(Color.WHITE);
        setFont(new Font("Segoe UI Symbol", Font.BOLD, 25));
    }

    @Override
    protected void paintComponent(Graphics g) {
        Graphics2D g2 = (Graphics2D) g;
        
        GradientPaint degradado = new GradientPaint(0, 0, color1,
            0, getHeight(), color2);
            
        g2.setPaint(degradado);
        g2.fillRoundRect(0, 0, getWidth(), getHeight(), 20, 20);
        
        super.paintComponent(g);
    }
}

```

Como puedes ver, los colores se han parametrizado como constantes del botón base (`BotonBonito`).

### 8.1.2. Añadir botones al panel

El objeto `jpbotones` se ve demasiadas veces en y siempre para hacer lo mismo: incorporar un botón.

Podemos rebajar esta incesante masa de líneas de código con un método que lo haga más sencillo. Para ello añadimos a nuestra clase `ViewCalculadora.java` (recuerda que en el paso anterior hemos renombrado `interfaz_calculadora.java`) el siguiente método:

```java
...
    /**
     * Este método añade al panel indicado los componentes que se pasan
     * como argumento variádico (equivalente a un array de elementos
     * JComponent).
     * @param panel Panel al que añadir los distintos componentes.
     * @param componentes Componentes a añadir.
     */
    private void addToPanel(JPanel panel, JComponent... componentes) {
        for(JComponent componente : componentes) {
            panel.add(componente);
        }
    }
...
```

> [!note] Nota 
> Recordemos que todos los componentes heredan de `JComponent`, por lo que podemos pasar botones, etiquetas, *radio buttons*, etc.

Así, para añadir los botones de números y operaciones a  `jpbotones`, en lugar de hacer esto:

```java
...
        jpbotones.add(btRaiz);
        jpbotones.add(btDivison);
        jpbotones.add(btBorrar);
        jpbotones.add(btBorrarTodo);

        jpbotones.add(btNum7);
...
```

Con este método se puede reducir a lo siguiente:

```java
...

                btRaiz, btDivison, btBorrar, btBorrarTodo,
                btNum7, btNum8, btNum9, btMultiplicar,
                btNum4, btNum5, btNum6, btRestar,
                btNum1, btNum2, btNum3, btSumar,
                btModulo, btNum0, btComa, btIgual);
...
```

Y se añaden así para respetar el orden en que aparecerán en pantalla.

**De 20 líneas a 1** (o a 6, porque hemos partido la línea para que se vea sin necesidad de hacer scroll horizontal).

```java
/**
 * Establece las características básicas que deben tener todos los botones

 * No se puede instanciar directamente, ya que es abstracta.
 */
public abstract class BotonBonito extends JButton {
    public static final Color COLOR_OPER1 = new Color(80, 80, 80);
    public static final Color COLOR_OPER2 = new Color(30, 30, 30);
    public static final Color COLOR_NUMERO1 = new Color(30, 180, 150);
    public static final Color COLOR_NUMERO2 = new Color(10, 120, 100);

    private final Color color1;
    private final Color color2;

    public BotonBonito(String texto, Color c1, Color c2) {
        super(texto);
        this.color1 = c1;
        this.color2 = c2;

        setFocusPainted(false);
        setContentAreaFilled(false);
        setBorderPainted(true);
        setForeground(Color.WHITE);
        setFont(new Font("Segoe UI Symbol", Font.BOLD, 25));
    }

    @Override
    protected void paintComponent(Graphics g) {
        Graphics2D g2 = (Graphics2D) g;

        GradientPaint degradado = new GradientPaint(0, 0, color1,
                0, getHeight(), color2);

        g2.setPaint(degradado);
        g2.fillRoundRect(0, 0, getWidth(), getHeight(), 20, 20);

        super.paintComponent(g);
    }
}
```

## 8.2. Aplicar MVC

Este patrón sirve para desacoplar la interfaz de usuario (View) y la del modelo de datos (Model) usando una capa para comunicar ambas (Controller). De esta forma, si una de esas capas cambia, no hay que preocuparse en exceso de los errores en cascada que puedan surgir.

Para verlo en la práctica, vamos a usar un ejemplo sencillo (se ha extraído de [Google Gemini](https://gemini.google.com/)). El siguiente código muestra por consola un menú para incrementar o decrementar un contador:

`AplicacionContador.java`
```java
import java.util.Scanner;

public class AplicacionContador {
    // El único dato que maneja: el contador
    private int contador = 0;

    public void iniciar() {
        Scanner scanner = new Scanner(System.in);
        String opcion = "";

        while (!opcion.equals("3")) {
            // INTERFAZ (Vista oculta)
            System.out.println("\n--- CONTADOR MONOLÍTICO ---");
            System.out.println("Valor actual: " + contador);
            System.out.println("1. Incrementar");
            System.out.println("2. Decrementar");
            System.out.println("3. Salir");
            System.out.print("Selecciona una opción: ");

            opcion = scanner.nextLine();

            // LÓGICA Y ENRUTAMIENTO (Controlador oculto)
            if (opcion.equals("1")) {
                contador++;
            } else if (opcion.equals("2")) {
                contador--;
            }
        }
        scanner.close();
        System.out.println("¡Adiós!");
    }

    public static void main(String[] args) {
        new AplicacionContador().iniciar();
    }
}
```

Como puede verse, toda la aplicación (interfaz de usuario, lógica y datos) está en un solo fichero.
Para aplicar el patrón de diseño MVC, tenemos que diferenciar de forma clara las tres capas:

- **Modelo:** guarda los datos puros y las reglas de negocio, sin incluir en él absolutamente **nada que tenga que ver con la pantalla**.
- **Vista:** muestra los datos en pantalla y captura las acciones del usuario.
- **Controlador:** es el cerebro que une a ambos. Recibe la acción de la vista (View), envía mensajes al modelo para cambiar datos y actualiza la Vista con los datos que le proporciona el modelo.

Para separar estas capas en nuestro proyecto, podemos crear la siguiente estructura de paquetes:

```
├── controller
│   ├── ContadorController.java
├── model
│   ├── ContadorModel.java
└── view
│   └── ContadorView.java
└── Main.java
```

### 8.2.1. Modelo

Esta aplicación maneja solo un dato: el contador.
El modelo se encarga de gestionar los datos de la aplicación, así que en este caso será el responsable de guardarlo (como una propiedad) y de ofrecer interfaces (métodos) de todas las operaciones que se pueden realizar sobre estos datos:
```java
public class ContadorModel {
    private int valor = 0;

    public int getValor() {
        return valor;
    }

    public void incrementar() {
        valor++;
    }

    public void decrementar() {
        valor--;
    }
}
```

### 8.2.2. Vista

Se encarga de mostrar por pantalla lo que el usuario ve, ¡nada más! Es decir, muestra los datos al usuario por la consola y lee de ella lo que el usuario indique.

```java
import java.util.Scanner;

public class ContadorView {
    private Scanner scanner = new Scanner(System.util.System.in);

    // Muestra el estado del modelo que le pasa el controlador
    public void mostrarContador(int valor) {
        System.out.println("\n--- CONTADOR MVC ---");
        System.out.println("Valor actual: " + valor);
        System.out.println("1. Incrementar");
        System.out.println("2. Decrementar");
        System.out.println("3. Salir");
        System.out.print("Selecciona una opción: ");
    }

    public String leerOpcion() {
        return scanner.nextLine();
    }

    public void mostrarMensaje(String mensaje) {
        System.out.println(mensaje);
    }
}

```

### 8.2.3. Controlador

Coordina el flujo entre la vista y el modelo.

```java
public class ContadorController {
    private ContadorModel model;
    private ContadorView vista;

    // Recibe las referencias por constructor
    public ContadorController(ContadorModel model, ContadorView vista) {
        this.model = model;
        this.vista = vista;
    }

    public void iniciarAplicacion() {
        String opcion = "";

        while (!opcion.equals("3")) {
            // 1. Le pide a la vista que dibuje usando los datos del modelo
            vista.mostrarContador(model.getValor());
            
            // 2. Escucha la acción del usuario
            opcion = vista.leerOpcion();

            // 3. Modifica el modelo según la acción de la vista
            if (opcion.equals("1")) {
                model.incrementar();
            } else if (opcion.equals("2")) {
                model.decrementar();
            }
        }
        vista.mostrarMensaje("¡Adiós desde MVC!");
    }
}
```

Como podemos ver, el controlador tiene como propiedades una instancia de la vista y otra del modelo, de forma que puede acceder ambos e intercambiar datos entre ellos.

> [!note] Nota 
> Se podrían inicializar el modelo y la vista en la declaración, y así no sería necesario pasarlos en el constructor.
> ```java
> public class ContadorController {
>     private ContadorModel modelo = new ContadorModel();
>     private ContadorView vista = new ContadorView();
>     ...
> }
> ```

### 8.2.4. Main

Para ejecutar nuestra aplicación basada en MVC hay que ejecutar el controlador que, como hemos visto, se encargará de iniciar la vista y el modelo:

```java
public class Main {
    public static void main(String[] args) {
        // Instanciar los tres componentes
        ContadorModel modelo = new ContadorModel();
        ContadorView vista = new ContadorView();
        
        // El controlador conecta el modelo y la vista
        ContadorController controlador = new ContadorController(
                new ContadorModel(),
                new ContadorView()
        );
        
        // Arrancar
        controlador.iniciarAplicacion();
    }
}
```
