```java
import javax.swing.*;
import java.awt.*;

/**
 * Esta clase extiende de JButton para crear botones personalizados
 * 
 * @author Rubén J. M.
 */
public class BotonBonito extends JButton {

    /*
    Aributos de la clase que vamos a usar para crear botones
    personalizados con degragadados
    */
    private Color color1;
    private Color color2;

    /*
    Constructor que recibe el texto del boton y los 2 colores
    para el degradado
    */
    public BotonBonito(String texto, Color c1, Color c2) {
        super(texto);
        this.color1 = c1;
        this.color2 = c2;

        setFocusPainted(false);
        setContentAreaFilled(false);
        setBorderPainted(true);
        setForeground(Color.WHITE);
        setFont(new Font("Segoe UI Symbol", Font.BOLD, 25));
    }
    /*
    Sobreescribe el metodo paintComponent para pintar el degradado
    primero pinta el fondo con degradado y luego llama al super
    para el texto
    */
    @Override
    protected void paintComponent(Graphics g) {
        Graphics2D g2 = (Graphics2D) g;

        GradientPaint degradado = new GradientPaint(0, 0, color1, 0, getHeight(), color2);

        g2.setPaint(degradado);
        g2.fillRoundRect(0, 0, getWidth(), getHeight(), 20, 20);

        super.paintComponent(g);
    }
}
```

```java
    /**
     * calcula de raiz cuadrada

    public void calcularRaiz() {
        double numero = Double.parseDouble(textoActual());
        mostrarResultado(Math.sqrt(numero));

    /**
     * Este método comprueba si ya tiene una coma antes de ponerla
     * y si no la tiene la pone y lo concatena a lo q ya había
     */
    public void añadirComa() {
        if (!textoActual().contains(".")) {
            jtpantalla.setText(textoActual() + ".");

    /**
     * Este método se encarga de borrar el último número de la
     * pantalla, lo borra a través del .substrinng el cual con
     * los parámetros (0, textoActual().lenght-1)
     * indicas que quieres quedarte en este caso indica que quiero

     * penúltimo), también chequea si el texto actual está vacío
     * si lo está reinicia toda la operación y borra el contenido

    public void borrarUltimo() {
        if (!textoActual().isEmpty()) {
            jtpantalla.setText(textoActual().substring(0, textoActual().length() - 1));

                concatenarNumero("1");
                break;
            case "2":
                concatenarNumero("2");
                break;
            case "3":
                concatenarNumero("3");
                break;
            case "4":
                concatenarNumero("4");
                break;
            case "5":
                concatenarNumero("5");
                break;
            case "6":
                concatenarNumero("6");
                break;
            case "7":
                concatenarNumero("7");
                break;
            case "8":
                concatenarNumero("8");
                break;
            case "9":
                concatenarNumero("9");
                break;

            case "+":
                guardarOperador("+");
                break;
            case "-":
                guardarOperador("-");
                break;
            case "/":
                guardarOperador("/");
                break;
            case "*":
                guardarOperador("*");
                break;
            case "%":
                guardarOperador("%");
                break;

            // llamamos al método calcular el cual se encarga de hacer la cuenta
            // correspondiente al operador guardado anteriormente
            case "=":
                calcular();
                break;

            case "C":
                limpiarPantalla();
                break;

            case "⌫":
                borrarUltimo();
                break;

            case "√":
                calcularRaiz();
                break;

            case ",":
                añadirComa();
                break;

            default:
                break;
        }
    }

    /**
     * Este método lo que hace es un switch donde segun el operador que hemos
     * guardado con el método guardar operador
     * recuperamos ese valor de la variable auxiliar y lo usamos para decidir la
     * operación, además usamos un método que es
     * mostrarResultado el cual muestra en la pantalla el resultado
     */
    public void calcular() {
        double numero2 = Double.parseDouble(textoActual());
        double resultado;

        switch (operador) {
            case "+":
                resultado = numero1 + numero2;
                mostrarResultado(resultado);
                break;
            case "-":
                resultado = numero1 - numero2;
                mostrarResultado(resultado);
                break;
            case "*":
                resultado = numero1 * numero2;
                mostrarResultado(resultado);
                break;
            case "/":
                /*
                con este if sabemos si el número 2 es 0 entonces
                directamente no hacemos la división, si no es el
                caso hacemos la división como de costumbre
                */
                if (numero2 == 0) {
                    jtpantalla.setText("Error: div/0");
                } else {
                    resultado = numero1 / numero2;
                    mostrarResultado(resultado);
                }

                break;
            // me calcula el resto de la división de los números
            case "%":
                resultado = numero1 % numero2;
                mostrarResultado(resultado);
                break;

            default:
                break;
        }
    }
```

### 8.1.1. Aplicar reglas de estilo básicas

Antes de nada, es conveniente seguir unas reglas o convenciones y que se apliquen uniformemente en toda la aplicación.

> [!example] Repaso 
> Recuerda las normas de estilo que vimos en el apartado [[6. Reglas de estilo]] de la [[Unidad 2 - Estructuras de control. Calidad del software|unidad 2]].

#### Renombrar clase

Vamos a **renombrar `interfaz_calculadora.java` como `ViewCalculadora.java`**. Aún así, es conveniente que se apliquen estas reglas a todo el código: nombres de variables, constantes, métodos, clases...

#### Renombrar clase
Evitar _hardcodear_ valores en el código. En su lugar, parametrizarlos como **constantes** (por ejemplo) o llevarlos a **enumerados** (como veremos más adelante). 

En este caso, lo haremos con los signos de operaciones (el resto te lo dejamos a ti). Refactorizamos haciendo click derecho del ratón sobre un literal como `"+"` y elegimos la opción de refactorizar para pasarlo a una constante. Así, acabaremos teniendo algo así:


Y no usaremos esas constantes en el código.

### 8.1.2. Botones

Para empezar, podemos simplificar la creación de botones. Hasta ahora vemos que para cada botón hay que introducir muchos parámetros y esto puede llevar a confusión y posibles errores.

Pero si creamos una clase botón para los números y otra para las operaciones, podríamos tener lo siguiente:

`BotonBonito.java`
```java
package view;

import javax.swing.*;
import java.awt.*;

/**
 * Establece las características básicas que deben tener todos los botones
 * de la aplicación.
 * No se puede instanciar directamente, ya que es abstracta.
 */
public abstract class BotonBonito extends JButton {
    public static final Color COLOR_OPER1 = new Color(80, 80, 80);
    public static final Color COLOR_OPER2 = new Color(30, 30, 30);
    public static final Color COLOR_NUMERO1 = new Color(30, 180, 150);
    public static final Color COLOR_NUMERO2 = new Color(10, 120, 100);

    private final Color color1;
    private final Color color2;

    public BotonBonito(String texto, Color c1, Color c2) {
        super(texto);
        this.color1 = c1;
        this.color2 = c2;

        setFocusPainted(false);
        setContentAreaFilled(false);
        setBorderPainted(true);
        setForeground(Color.WHITE);
        setFont(new Font("Segoe UI Symbol", Font.BOLD, 25));
    }

    @Override
    protected void paintComponent(Graphics g) {
        Graphics2D g2 = (Graphics2D) g;
        
        GradientPaint degradado = new GradientPaint(0, 0, color1,
            0, getHeight(), color2);
            
        g2.setPaint(degradado);
        g2.fillRoundRect(0, 0, getWidth(), getHeight(), 20, 20);
        
        super.paintComponent(g);
    }
}

```

Como puedes ver, los **colores se han parametrizado como constantes** del botón base (`BotonBonito`). Además, **la clase se ha puesto como `abstract`** para evitar el libre albedrío en la creación de botones.

> [!question] ¿Cómo lo harías? 🤔
> En lugar de constantes de una clase, podríamos usar un enum llamado `ColorBoton` que aglutinase todos los colores de los botones.

### 8.1.3. Añadir botones al panel

El objeto `jpbotones` se ve demasiadas veces en y siempre para hacer lo mismo: incorporar un botón.

Podemos rebajar esta incesante masa de líneas de código con un método que lo haga más sencillo. Para ello añadimos a nuestra clase `ViewCalculadora.java` el siguiente método:

```java
...
    /**
     * Este método añade al panel indicado los componentes que se pasan
     * como argumento variádico (equivalente a un array de elementos
     * JComponent).
     * @param panel Panel al que añadir los distintos componentes.
     * @param componentes Componentes a añadir.
     */
    private void addToPanel(JPanel panel, JComponent... componentes) {
        for(JComponent componente : componentes) {
            panel.add(componente);
        }
    }
...
```

> [!note] Nota 
> Recordemos que todos los componentes heredan de `JComponent`, por lo que podemos pasar botones, etiquetas, *radio buttons*, etc.

Así, para añadir los botones de números y operaciones a  `jpbotones`, en lugar de hacer esto:

```java
...
        jpbotones.add(btRaiz);
        jpbotones.add(btDivison);
        jpbotones.add(btBorrar);
        jpbotones.add(btBorrarTodo);

        jpbotones.add(btNum7);
...
```

Con este método se puede reducir a lo siguiente:

```java
...

                btRaiz, btDivison, btBorrar, btBorrarTodo,
```

### 8.1.4. Asignar action listener a los botones

```java
...

     * A todos los botones introducidos como parámetros se les asigna
     * este objeto como gestor de eventos (ActionListener).
     * @param botones
```

#### `calcular()`

En este método se repite mucho código en el `switch`.  Una de las ventajas del nuevo `switch` es que puede usarse como una **expresión**. Es decir, puede ponerse donde se espera un valor de un tipo concreto. Por eso podemos replantear este código:

```java
public void calcular() {
	double numero2 = Double.parseDouble(textoActual());
	double resultado;
	
	switch (operador) {
		case SUMA:
			resultado = numero1 + numero2;
			mostrarResultado(resultado);
			break;
		case RESTA:
			resultado = numero1 - numero2;
			mostrarResultado(resultado);
			break;
		case MULTIPLICACION:
			resultado = numero1 * numero2;
			mostrarResultado(resultado);
			break;
		case DIVISION:
		    /*
		    con este if sabemos si el número 2 es 0 entonces
		    directamente no hacemos la división, si no es el
		    caso hacemos la división como de costumbre
		    */
			if (numero2 == 0) {
				jtpantalla.setText("Error: div/0");
			} else {
				resultado = numero1 / numero2;
				mostrarResultado(resultado);
			}
			break;
		// me calcula el resto de la división de los números
		case RESTO:
			resultado = numero1 % numero2;
			mostrarResultado(resultado);
			break;

		default:
			break;
	}
}
```

Para simplificarlo de esta manera:

```java
public void calcular() {
	double numero2 = Double.parseDouble(textoActual());
	
	// Cláusula guardián
	if (operador == DIVISION && numero2 == 0) {
		jtpantalla.setText("Error: div/0");
		return;
	}

	// Usemos las posibilidades del nuevo switch:
	double resultado = switch (operador) {
		case SUMA -> numero1 + numero2;
		case RESTA -> numero1 - numero2;
		case MULTIPLICACION -> numero1 * numero2;
		case DIVISION -> numero1 / numero2;
		case RESTO -> numero1 % numero2;
		default -> throw new IllegalArgumentException("Error! Operador no válido: " + operador);
	};
	mostrarResultado(resultado);
}
```

El código es más claro, menos extenso... en definitiva, más fácil de mantener y de arreglar si hubiese errores.

Observa esa cláusula guardián al principio. Si se trata de una división por cero, directamente muestra un mensaje en el display de la calculadora y sale del método. Fin.

Y por otro lado, a la variable `resultado` se le asigna el resultado del `switch`, y que este devuelve una de las operaciones aritméticas. Si no, si el operador no es correcto, hace saltar una excepción.

La aplicación de nuestro ejemplo está, en su mayoría, en el fichero donde creamos la ventana.

Para verlo en la práctica, vamos a usar un ejemplo sencillo (se ha extraído de [Google Gemini](https://gemini.google.com/)). El siguiente código muestra por consola un menú para incrementar o decrementar un contador:

`AplicacionContador.java`
```java
import java.util.Scanner;

public class AplicacionContador {
    // El único dato que maneja: el contador
    private int contador = 0;

    public void iniciar() {
        Scanner scanner = new Scanner(System.in);
        String opcion = "";

        while (!opcion.equals("3")) {
            // INTERFAZ (Vista oculta)
            System.out.println("\n--- CONTADOR MONOLÍTICO ---");
            System.out.println("Valor actual: " + contador);
            System.out.println("1. Incrementar");
            System.out.println("2. Decrementar");
            System.out.println("3. Salir");
            System.out.print("Selecciona una opción: ");

            opcion = scanner.nextLine();

            // LÓGICA Y ENRUTAMIENTO (Controlador oculto)
            if (opcion.equals("1")) {
                contador++;
            } else if (opcion.equals("2")) {
                contador--;
            }
        }
        scanner.close();
        System.out.println("¡Adiós!");
    }

    public static void main(String[] args) {
        new AplicacionContador().iniciar();
    }
}
```

Como puede verse, toda la aplicación (interfaz de usuario, lógica y datos) está en un solo fichero.
Para aplicar el patrón de diseño MVC, tenemos que diferenciar de forma clara las tres capas:

- **Modelo:** guarda los datos puros y las reglas de negocio, sin incluir en él absolutamente **nada que tenga que ver con la pantalla**.
- **Vista:** muestra los datos en pantalla y captura las acciones del usuario.
- **Controlador:** es el cerebro que une a ambos. Recibe la acción de la vista (View), envía mensajes al modelo para cambiar datos y actualiza la Vista con los datos que le proporciona el modelo.

Para separar estas capas en nuestro proyecto, podemos crear la siguiente estructura de paquetes:

```
├── controller
│   ├── ControllerCalculadora.java
├── model
│   ├── ModelCalculadora.java
└── view
│   ├── BotonBonito.java
│   ├── BotonNumero.java
```

### 8.2.1. Modelo

Esta aplicación manejará los datos del **primer operando** (`numero1`) y del **resultado** (`resultado`, aunque este último no sería necesario incluirlo en el modelo).

¿Por qué no guardamos el segundo operando? Por varios motivos:
1. Si la operación es _unaria_ (solo necesita un operando, como en la raíz cuadrada), una vez realizada la operación, **el resultado pasa a ser el primer operando** por si se siguen introduciendo operaciones sobre ese valor.
2. Se obtiene solo para realizar la operación y, acto seguido, se desprecia. Puedes verlo en el código original. En `interfaz_calculadora.java` tenemos la propiedad `numero1` pero no la `numero2`.

En conclusión, nuestro modelo quedaría de la forma siguiente:

`ModelCalculadora.java`

```java
package model;

/**
 * Recordemos que el modelo debe ocuparse solo de los datos y la lógica de negocio
 * que los trata. ¡Nada más!
 * <p>
 * Debe permitir el manejo y la recuperación de los datos desde fuera.
 * <p>
 * Nunca se debe llamar desde el modelo a la vista para mostrar datos. Así, si la
 * vista se estropea por lo que sea, el modelo no va detrás (errores en cascada).
 * Es decir, evitamos el acoplamiento.
 * <p>
 * En nuestro caso, el modelo guardará el primer operando (numero1) y el resultado.
 * No es necesario (para este ejemplo) guardar numero2 ya que será borrado
 * inmediatamente después del cálculo.
 * @author Rubén J. M.
 * @author Diego G.S.
 */
public class ModelCalculadora {
    private double numero1 = 0;
    private double resultado = 0;

    /**
     * Al resetear el primer operando, se va a comenzar una nueva
     * operación. Por tanto, el resultado también se resetea.
     */
    public void resetNumero1() {
        this.numero1 = 0;
        this.resultado = 0;
    }

    public double getResultado() {
        return this.resultado;
    }

    public double getNumero1() {
        return numero1;
    }

    public void setNumero1(double numero1) {
        this.numero1 = numero1;
    }

    public void setNumero1(String numero1) {
        try {
            this.numero1 = Double.parseDouble(numero1);
        } catch (NumberFormatException e) {
            throw new IllegalArgumentException("Error! Número no válido");
        }
    }

    // Lógica del modelo
    public void sumar(double numero2) {
        resultado = numero1 + numero2;
    }

    public void restar(double numero2) {
        resultado = numero1 - numero2;
    }

    public void multiplicar(double numero2) {
        resultado = numero1 * numero2;
    }

    public void dividir(double numero2) {
        if (numero2 == 0)
            throw new IllegalArgumentException("Erro! División por 0");
        resultado = numero1 / numero2;
    }

    public void resto(double numero2) {
        resultado = numero1 % numero2;
    }

    public void raizCuadrada() {
        resultado = Math.sqrt(numero1);
    }
}
```

> [!warning] Recuerda... 
> La intención es conseguir el **desacoplamiento entre vista y modelo**. Si ves el código anterior, en ningún momento hay referencias a la ventana ni a ningún otro elemento de la aplicación.

### 8.2.3. Controlador

Es el controlador el que tiene que tener acceso a la vista y al modelo, además de las funcionalidades adecuadas para comunicar a ambos:

```java
package controller;

import model.ModelCalculadora;
import view.BotonNumero;
import view.BotonOperacion;
import view.ViewCalculadora;

import javax.swing.*;
import java.awt.*;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;

/**
 * El controller comunica la vista y el modelo.
 * Para ello debe tener una referencia a cada uno de ellos y
 * acceder a sus propiedades y métodos públicos para gestionarlos.
 * @author Rubén J. M.
 * @author Diego G.S.
 */
public class ControllerCalculadora implements ActionListener {
    private ModelCalculadora model = new ModelCalculadora();
    private ViewCalculadora view = new ViewCalculadora();

    public static final String CERO = "0";
    public static final String UNO = "1";
    public static final String DOS = "2";
    public static final String TRES = "3";
    public static final String CUATRO = "4";
    public static final String CINCO = "5";
    public static final String SEIS = "6";
    public static final String SIETE = "7";
    public static final String OCHO = "8";
    public static final String NUEVE = "9";

    public static final String SUMA = "+";
    public static final String RESTA = "-";
    public static final String DIVISION = "/";
    public static final String MULTIPLICACION = "*";
    public static final String RESTO = "%";
    public static final String COMA = ",";
    public static final String IGUAL = "=";
    public static final String LIMPIAR = "C";
    public static final String BORRAR = "⌫";
    public static final String RAIZ_CUADRADA = "√";
    public static final String PUNTO_DECIMAL = ".";

    private boolean operacionRealizada = false;
    private String operador = "";
    private double numero1;

    public ControllerCalculadora() {
        setVisualEnvironment();
    }

    /**
     * Crea y configura la ventana y los componentes visuales (botones,
     * etiquetas, etc.).
     * Una vez hecho, muestra la pantalla.
     */
    private void setVisualEnvironment() {
        setVentana();
        setDisplay();
        setButtons();
        view.setVisible(true);
    }

    /**
     * <p>
     *     Establece las características de la ventana de la calculadora:
     *     <ul>
     *         <li>El título</li>
     *         <li>Las dimensiones</li>
     *         <li>El layout manager</li>
     *         <li>Los paneles que contiene</li>
     *     </ul>
     * </p>
     *
     *
     */
    private void setVentana() {
        view.setTitle("Calculadora");
        view.setLayout(new BorderLayout(5, 10));
        view.setSize(470, 600);
        view.setLocationRelativeTo(null); // para situar la ventana centro de la pantalla
        view.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);

        // Añadimos los paneles del display y los botones
        view.panelPantalla = new JPanel(new FlowLayout());
        view.panelBotones = new JPanel(new GridLayout(5, 4));

        view.add(view.panelPantalla, BorderLayout.NORTH);
        view.add(view.panelBotones, BorderLayout.CENTER);
    }

    /**
     * Establece las características del display de la calculadora donde se muestran
     * los números, operaciones y resultados.
     */
    public void setDisplay() {
        view.txtPantalla = new JTextField(15);
        view.txtPantalla.setFont(new Font("Arial", Font.BOLD, 30));
        view.txtPantalla.setHorizontalAlignment(JTextField.RIGHT);
        view.txtPantalla.setEditable(false);

        view.panelPantalla.add(view.txtPantalla);
        view.txtPantalla.setBackground(Color.DARK_GRAY);
        view.txtPantalla.setForeground(Color.WHITE);
        view.txtPantalla.setBorder(BorderFactory.createEmptyBorder(1, 1, 1, 1));

    }

    /**
     * Crea los botones, los añade a los paneles correspondientes y
     * les asigna el ActionListener.
     */
    public void setButtons() {
//        List<JButton> botonesNumericos = Arrays.asList();
        view.btModulo = new BotonOperacion(RESTO);
        view.btSumar = new BotonOperacion(SUMA);
        view.btRestar = new BotonOperacion(RESTA);
        view.btMultiplicar = new BotonOperacion(MULTIPLICACION);
        view.btDivison = new BotonOperacion(DIVISION);
        view.btRaiz = new BotonOperacion(RAIZ_CUADRADA);
        view.btBorrar = new BotonOperacion(BORRAR);
        view.btComa = new BotonOperacion(COMA);
        view.btIgual = new BotonOperacion(IGUAL);
        view.btBorrarTodo = new BotonOperacion(LIMPIAR);

        view.btNum0 = new BotonNumero(CERO);
        view.btNum1 = new BotonNumero(UNO);
        view.btNum2 = new BotonNumero(DOS);
        view.btNum3 = new BotonNumero(TRES);
        view.btNum4 = new BotonNumero(CUATRO);
        view.btNum5 = new BotonNumero(CINCO);
        view.btNum6 = new BotonNumero(SEIS);
        view.btNum7 = new BotonNumero(SIETE);
        view.btNum8 = new BotonNumero(OCHO);
        view.btNum9 = new BotonNumero(NUEVE);

        addToPanel(view.panelBotones,
                view.btRaiz, view.btDivison, view.btBorrar, view.btBorrarTodo,
                view.btNum7, view.btNum8, view.btNum9, view.btMultiplicar,
                view.btNum4, view.btNum5, view.btNum6, view.btRestar,
                view.btNum1, view.btNum2, view.btNum3, view.btSumar,
                view.btModulo, view.btNum0, view.btComa, view.btIgual);

        setThisActionListenerTo(
                view.btNum0, view.btNum1, view.btNum2, view.btNum3, view.btNum4,
                view.btNum5, view.btNum6, view.btNum7, view.btNum8, view.btNum9,
                view.btRaiz, view.btDivison, view.btBorrar, view.btBorrarTodo,
                view.btMultiplicar, view.btRestar, view.btSumar, view.btIgual,
                view.btModulo, view.btComa);
    }

    /**
     * Este método añade al panel indicado los componentes que se pasan
     * como argumento variádico (equivalente a un array de elementos
     * JComponent).
     *
     * @param panel       Panel al que añadir los distintos componentes.
     * @param componentes Componentes a añadir.
     */
    private void addToPanel(JPanel panel, JComponent... componentes) {
        for (JComponent componente : componentes) {
            panel.add(componente);
        }
    }

    /**
     * A todos los botones introducidos como parámetros se les asigna
     * este objeto como gestor de eventos (ActionListener).
     *
     * @param botones
     */
    private void setThisActionListenerTo(JButton... botones) {
        for (JButton boton : botones) {
            boton.addActionListener(this);
        }
    }

    /**
     * Limpia el display de la calculadora.
     */
    private void clearDisplay() {
        view.txtPantalla.setText("");
    }

    /**
     * Devuelve el texto del display.
     *
     * @return
     */
    private String displayValue() {
        return view.txtPantalla.getText();
    }

    /**
     * Resetea el estado de la calculadora para comenzar otra operación.
     * Es decir, reinicia el primer operando y el operador.
     */
    public void reset() {
        clearDisplay();

        // Reinicia el modelo.
        model.resetNumero1();

        operador = "";
        operacionRealizada = false;
    }

    /**
     * Añade el dígito introducido a los que ya hay en el display.
     * Utiliza el flag operacionRealizada para saber si es una nueva
     * operación o no.
     *
     * @param numero Nuevo dígito a mostrar.
     */
    public void addDigitToDisplay(String numero) {
        if (operacionRealizada) {
            clearDisplay();
            operacionRealizada = false;
        }
        view.txtPantalla.setText(displayValue() + numero);
    }

    public void setOperacion(String op) {
        if (!operador.isEmpty() && !operacionRealizada) {
            // si ya hay un operador guardado hace el cálculo
            calcular();
        }
        // Los datos los gestiona el modelo:
        try {
            model.setNumero1(displayValue());
        }
        catch (NumberFormatException e) {
            /*
            Gestionamos la NumberFormatException que lanzamos en
            el método setNumero1. Si salta, mostramos por pantalla
            el mensaje "Error! Número no válido" que indicamos.
            */
            view.txtPantalla.setText(e.getMessage());
        }

        // guarda el operador en una variable auxiliar
        operador = op;

        clearDisplay();
    }

    /**
     * Pide al modelo que realice la operación aritmética (suma, resta,
     * multiplicación, división o resto) del primer operando con el segundo.
     * Es una operación binaria de cálculo, ya que usa dos operandos (numero1
     * y numero2).
     * También muestra el resultado en el display.
     */
    public void calcular() {
        double numero2 = Double.parseDouble(displayValue());

        if (operador.equals(DIVISION) && numero2 == 0) {
            reset();
            view.txtPantalla.setText("Error: div/0");
            return;
        }

        switch (operador) {
            case SUMA -> model.sumar(numero2);
            case RESTA -> model.restar(numero2);
            case MULTIPLICACION -> model.multiplicar(numero2);
            case DIVISION -> model.dividir(numero2);
            case RESTO -> model.resto(numero2);
            default -> {
                reset();
                view.txtPantalla.setText("Error! Operador no válido: " + operador);
            }
        }
        mostrarResultado();
    }

    /**
     * Borrar el último dígito del display.
     */
    public void removeLastDigit() {
        if (displayValue().length() <= 1) {
            reset();
            return;
        }
        view.txtPantalla.setText(displayValue().substring(0, displayValue().length() - 1));
    }

    /**
     * Muestra en el display el resultado, que se obtiene del modelo.
     * También cambia el flag para indicar si se realiza o no la operación
     */
    public void mostrarResultado() {
        double resultado = model.getResultado();
        view.txtPantalla.setText(Double.toString(resultado));
        operacionRealizada = true;
    }


    /**
     * Añade la coma (o punto, en nomenclatura inglesa) decimal,
     * si no había una antes.
     */
    public void addComa() {
        if (!displayValue().contains(PUNTO_DECIMAL)) {
            view.txtPantalla.setText(displayValue() + PUNTO_DECIMAL);
        }
    }

    /**
     * Tengamos presenta que si no se le asigna ningún "action name" a un botón,
     * asume automáticamente el texto que tenga.
     *
     * @param e the event to be processed
     */
    @Override
    public void actionPerformed(ActionEvent e) {
        String comando = e.getActionCommand();

        /*
        switch es más apropiado para revisar valores concretos.
        Si estamos evaluando rangos de valores, es mejor un if-else if-else.
        */
        if (comando.matches("\\d")) {
            addDigitToDisplay(comando);
        }
        else if (comando.matches("[+\\-/*%]")) {
            setOperacion(comando);
        }
        else if (comando.equals(RAIZ_CUADRADA)) {
            model.setNumero1(displayValue());
            model.raizCuadrada();
            mostrarResultado();
        }
        else if (comando.equals(IGUAL)) {
            calcular();
        }
        else if (comando.equals(LIMPIAR)) {
            reset();
        }
        else if (comando.equals(BORRAR)) {
            removeLastDigit();
        }
        else if (comando.equals(COMA)) {
            addComa();
        }
        else {
            view.txtPantalla.setText("Comando erróneo: " + comando);
        }
    }
}
```
